class Solution {
public:
    int removeElement(vector<int>& nums, int val) {
        int n = nums.size();
        if (n==0){
            return 0;
        }
        int fast = 0;
        int slow = 0;
        while(fast < n){
            if(nums[fast] == val){
                fast++;
            }
            else{
                nums[slow] = nums[fast];
                slow++;
                fast++;
            }
        }
        return slow;
    }
};
